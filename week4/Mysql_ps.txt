1-2. database 생성 및 사용(일종의 폴더 열기)
create database likelion;
use likelion;

2. Insert INTO users(id, name) values('1', 'seongbin');
- 명시를 안하면 다 작성해야함
->user()로 작성한다면 table 내부의 모든 변수에 값을 넣어야함

- 명시를 하면 필요한 내용만 작성하면 됨
-> user table 내부에 id, name외의 다른 변수가 있더라도 users(id, name)로 작성한다면
==>id, name을 set하려는 값만 작성 values('1', 'seongbin');

3. select id, name from users;
-from, where, select 순으로 읽힘

4. select * from users;
- *는 전부라는 의미

5. DROP table table명 || database database명 삭제
drop database mydatabase;
drop table mydata.user;
drop database aaaaa;
-단 table과 database가 생성되어 있어야 DROP 사용가능
